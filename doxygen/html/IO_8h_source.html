276 template <typename ParseVariableType>
277 ParseVariableType ReadNumberFromFile(std::string SettingName, std::string SettingCategory, const std::string& filePath)
278 {
279  if (!Utilities::fileExists(filePath)) { throw IOException(filePath + ": File not found!"); }
280 
281  std::string SettingValue = CIniFile::GetValue(SettingName, SettingCategory, filePath);
282  if (SettingValue.empty()) {
283  throw IOException("Variable: " + SettingName + " in Category: " + SettingCategory + " not found in File: " + filePath);
284  }
285 
286  char* endptr = nullptr;
287  ParseVariableType Value;
288  if (std::is_same<ParseVariableType, double>::value) { Value = strtod(SettingValue.c_str(), &endptr); }
289  if (std::is_same<ParseVariableType, int>::value) { Value = strtol(SettingValue.c_str(), &endptr, 0); }
290 
291  /* Error Handling for strtod or strtol functions*/
292  if (Value == -HUGE_VAL || Value == HUGE_VAL || endptr == SettingValue.c_str())
293  {
294  std::stringstream SettingParseError;
295  SettingParseError << std::endl;
296  SettingParseError << "Invalid value in Parameter File: " << filePath.c_str() << std::endl;
297  SettingParseError << "Category: " << SettingCategory.c_str() << std::endl;
298  SettingParseError << "Setting: " << SettingName.c_str() << std::endl;
299  SettingParseError << "Value: " << SettingValue.c_str() << std::endl;
300  throw std::runtime_error(SettingParseError.str());
301  }
302  return Value;
303 }
